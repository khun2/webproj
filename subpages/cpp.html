        #include 
        
        using namespace std;
                    
        int main() {
            ios::sync_with_stdio(false);
            cin.tie(0);
            long long n,q,l,r;
            long long max = 0;
            cin >> n >> q;
            vector v(n), cnt(n+1, 0);  
            for (long long i = 0; i < n; i++) {
                cin >> v[i];
            }
            for (long long i = 0; i < q; i++) {
                cin >> l >> r;
                cnt[l]++;
                if (r+1 < n+1) cnt[r+1]--;
            }
            for (long long i = 1; i < cnt.size(); i++) {
                cnt[i] += cnt[i-1];
            }
        
            sort(v.rbegin(), v.rend());
            sort(cnt.rbegin(), cnt.rend());

            for (long long i = 0; i < n; i++) {   
                max += v[i] * cnt[i];
            }
            cout << max;
        }
